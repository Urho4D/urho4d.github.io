If you render to texture you can draw it wherever you like. 

static QPixmap toPixmap(Image* image) {
     return QPixmap::fromImage(QImage{
                   image->GetData(),
                   image->GetWidth(),
                   image->GetHeight(),
                   QImage::Format_RGBA8888 });
}
